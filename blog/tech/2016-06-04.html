function getContent(src)
{
	var request=getHttpObject();
	if(request)
	{
		request.open("GET",src,true);
		request.onreadystatechange=function(){
			if(request.readyState==4)
					{
						//some codes
					}

		};
		request.send(null);
	}
	else
	{
		alert("error");		//错误处理
	}
}

function getHttpObject() 	//解决兼容性
{
	return new XMLHttpRequest();
}
